1.2. css
 <style>
 /*body设置*/
    body{
        margin: 0px;
        font-size: 14px;
        font-family: sans-serif;
    }
    * {
        box-sizing: border-box;
    }

    /*left-nav*/
    .left-nav{
        list-style-type: none; /*清空所有自带样式,包括li自带的大点样式*/
        position: fixed;
        padding: 0;
        margin: 0;
        background-color:#f1f1f1;
        width: 12%;
        height: 100%;
        overflow: auto;
    }
    .left-nav li a{
        display: block;
        padding: 12px 16px;
        color: black;
        text-decoration: none;
    }
    .left-nav li a:hover{
        background-color:#666666;
        color: #ffffff;
    }
    .left-nav li a.active{
        background-color: #a62d2d;
        color: #fff;
    }

    /*--cotent--*/
    .cotent{
        margin-left:12%;
        padding: 4px;
    }
    .cotent img{
       width: 20%;
        float: left;
        padding: 4px;
    }
</style>

2.2 css
 <style>
    body {
        margin: 0;
        font-size: 14px;
        font-family: sans-serif;
    }
    * {
        box-sizing: border-box;
    }
    .top-nav{
        list-style-type: none;
        margin: 0px;
        padding: 0px;
        background-color: #333333;
        width: 100%;
        overflow: hidden;
        position: fixed;
        top: 0;
        z-index: 1;
    }
    .top-nav li{
        float: left;
        border-right: 1px solid #999;
    }
    .top-nav li a{
        display: block;
        color: #fff;
        text-decoration: none;
        padding: 12px 16px;
        text-align: center;
    }

    .top-nav li a:hover{
        background-color: black;
    }
    .top-nav li a.active {
        background-color: #a62d2d;
    }

    .top-nav li.right{
        float: right;
    }

    /*--content--*/
    .content{
        margin-top:50px ;
    }
    .content img {
        width: 20%;
        float: left;
        padding: 4px;
    }
</style>
