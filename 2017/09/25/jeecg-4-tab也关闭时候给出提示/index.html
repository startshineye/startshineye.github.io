var closeTab = function (id) {
//如果关闭的是当前激活的TAB，激活他的前一个TAB
if ($("li.active").attr('id') == "tab_" + id) {
    $("#tab_" + id).prev().addClass('active');
    $("#" + id).prev().addClass('active');
}
//关闭TAB
$("#tab_" + id).remove();
$("#" + id).remove();
};

$(".nav-tabs").on("click", "[tabclose]", function (e) {

想要实现需求就需要在 $(“[addtabs]”).click()方法时候判断是否是指定的编辑页.

$(".nav-tabs").on("click", "[tabclose]", function (e) {
    id = $(this).attr("tabclose");

    /**************************yxm start****************************/
    var index = $(this).parent().html().indexOf("在线流程编辑");
    if(index>=0){
        var result = confirm("确定关闭吗？关闭前确认ivr流程已经保存");
        if(result){
             closeTab(id);
        }else{
            $(this).tab('show'); //此处使用return还是不能阻止继续执行,所以写了不存在的$(this).tab('show'); 让其报错停止
        }
    }else{
         closeTab(id);
    }
    /*************************yxm end**********************************/
});
