vue(25)-组件

前言

什么是组件：组件是Vue.js最强大的功能之一。组件可以扩展HTML元素，封装可重用的代码。在较高层面上，组件是自定义的元素，Vue.js的编译器为它添加特殊功能。在有些情况下，组件也可以是原生HTML元素的形式，以is特性扩展。组件就是一个大的对象

内容

1.组件可以看成是HTML的扩展:比如：可以看成是HTML的扩展，但是在HTML中没有,我们需要自己来扩展,调用,这样就引出了组件。
2.定义一个组件:
1.全局组件 

var Aaa = Vue.extend({
   template:"<h3>方式一:定义组件</h3>"
});
Vue.component("aaa",Aaa);

2.（就是把extend出对象放入component中)

1.全局组件

普通组件的使用:1.创建组件(Vue.extend方法创建一个组件): 2.注册组件(Vue.component方法注册组件)

1.1 定义
<body>
<div id = "box">
<!--3.调用组件-->
  <aaa></aaa>
</div>
<script>
/********
 * 1.方式一:定义组件(从Vue中extend出:内部是json结构)
 * *******/
var Aaa = Vue.extend({
   template:"<h3>方式一:定义组件</h3>"
});
/****************
 * 2.注册组件(注册组件时候取名)
 * ************************/
Vue.component("aaa",Aaa);
console.info("Aaa",Aaa);
var vm = new Vue({
 el:"#box",
    data:{
     msg:"hello vue"
    }
});
</script>
</body>

输出: 方式一:定义组件

1.2 组件里面放数据
var Aaa = Vue.extend({
    data(){
        return {
            msg:"组件里面放数据"
        }
    },
   template:"<h3>{{msg}}</h3>"
});
Vue.component("aaa",Aaa);

组件里面放数据：data(数据)必须是函数的形式,函数必须返回一个对象(一般为json对象) 

<body>
<div id = "box">
<!--3.调用组件-->
  <aaa></aaa>
</div>
<script>
/********
 * 1.方式一:定义组件(从Vue中extend出:内部是json结构)
 * *******/
var Aaa = Vue.extend({

        return {

        }
    },
   template:"<h3>{{msg}}</h3>"
});
/****************
 * 2.注册组件(注册组件时候取名)

 Vue.component("aaa",Aaa);
var vm = new Vue({
 el:"#box",

     msg:"hello vue"
    }

输出:组件里面放数据

1.3 组件里面放事件

点击组件时候组件里面的数据变化: Vue.extend()类似于new Vue()里面的方法定义相似:

<body>
<div id = "box">
<!--3.调用组件-->
  <aaa></aaa>
</div>
<script>
var Aaa = Vue.extend({
    data(){
        return {
            msg:"组件里面放数据"
        }
    },
    methods:{
        change:function(){
           this.msg="change事件执行";
        }
    },
   template:"<h3 @click='change'>{{msg}}</h3>"
});
Vue.component("aaa",Aaa);
var vm = new Vue({
 el:"#box",
    data:{
     msg:"hello vue"
    }
});
</script>
</body>

输出:

2. 局部组件

全局组件:我们把组件的注册从new Vue()中分离出来了,new Vue()其实就是一个根,我们把组件的注册放入到new Vue()对象中去，也就是可以放到vue中components里面。 

<body>
<div id = "box">
  <aaa></aaa>
</div>
<script>
var Aaa = Vue.extend({
    data(){
        return {
            msg:"局部组件"
        }
    },
   template:"<h3>{{msg}}</h3>"
});
var vm = new Vue({
 el:"#box",
    data:{
     msg:"hello vue"
    },
    components:{
      'aaa':Aaa
    }
});
</script>
</body>

输出:局部组件

3.组件的另一种方式

组件的另一种方式就是将组件的注册直接放入到组件中

1.全局组件: 

<body>
<div id = "box">
<!--3.调用组件-->
  <aaa></aaa>
</div>
<script>
Vue.component("aaa",{
    template:"组件另一种方式"
});
var vm = new Vue({
 el:"#box",
    data:{
     msg:"hello vue"
    }
});
</script>
</body>

输出:组件另一种方式

2.局部 

<body>
<div id = "box">
<!--3.调用组件-->
  <my-aaa></my-aaa>
</div>
<script>
var vm = new Vue({
 el:"#box",
    data:{
     msg:"hello vue"
    },
    components:{
        "my-aaa":{
            template:"局部组件另一种方式"
        }
    }
});
</script>
</body>

输出:局部组件另一种方式

4.动态组件

多个组件可以使用同一个挂载点，然后动态的在他们之间切换。使用保留的元素，动态地绑定到它的is特性。下面的列子在同一个vue实例下挂了aaa、bbb、ccc三个组件，通过特性currentView动态切换组件显示。其中 :is=”a” a指代是哪个组件

1.通过元素 2.由is决定是哪个组件 

<body>
<div id = "box">
  <input type="button" @click="a='aaa'" value="aaa组件">
  <input type="button" @click="a='bbb'" value="bbb组件">
  <component :is="a"></component>
</div>
<script>
var vm = new Vue({
  el:"#box",
   data:{
      a:"aaa"//指代组件名
   },
   components:{
      "aaa":{
          template:"<h2>我是组件aaa</h2>"
      },
       "bbb":{
          template:"<h2>我是组件bbb</h2>"
       }
   }
});
</script>
</body>

输出:

5.父子组件

父子节点调用:

<!--第一种:子组件名放在父节点里面-->
<aaa>
    <bbb></bbb>
</aaa>

调用时候:子组件直接在父组件的template里面 

<body>
<div id = "box">
  <!--第一种:子组件名放在父节点里面-->
   <aaa>
   </aaa>
</div>
<script>
var vm = new Vue({
  el:"#box",
   data:{
      a:"aaa"//指代组件名
   },
   components:{
      "aaa":{
          template:"<h2>我是组件aaa</h2><bbb></bbb>",
          components:{
              "bbb":{
                  template:"<h3>我是组件bbb</h3>"
              }
          }
      }
   }
});
</script>
</body>

父节点里面的数据,父节点可以访问,但是子节点不能访问。vue默认情况下,子组件也没法访问父组件数据 

<body>
<div id = "box">
  <!--第一种:子组件名放在父节点里面-->
   <aaa>
   </aaa>
</div>
<script>
var vm = new Vue({
  el:"#box",
   data:{
      a:"aaa"//指代组件名
   },
   components:{
      "aaa":{
          data(){
              return{
                  msg:"我是父节点数据"
              }
          },
          template:"<h2>我是组件aaa->{{msg}}</h2><bbb></bbb>",
          components:{
              "bbb":{
                  template:"<h3>我是组件bbb->{{msg}}</h3>"
              }
          }
      }
   }
 });
</script>
</body>

组件间传递数据:
使用模板标签:
1.子组件就想获取父组件data: props:[] 

<body>
<div id="box">
    <aaa>
    </aaa>
</div>

<template id="aaa">
    <h1>11111</h1>
    <bbb :mmm="msg2" :my-msg="msg"></bbb>
</template>
<script>
    var vm=new Vue({
        el:'#box',
        data:{
            a:'aaa'
        },
        components:{
            'aaa':{
                data(){
                    return {
                        msg:111,
                        msg2:'我是父组件的数据'
                    }
                },
                template:'#aaa',
                components:{
                    'bbb':{
                        props:{
                            'm':String,
                            'myMsg':Number
                        },
                        template:'<h3>我是bbb组件->{{mmm}} <br> {{myMsg}}</h3>'
                    }
                }
            }
        }
    });

</script>
</body>
